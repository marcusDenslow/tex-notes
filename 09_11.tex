\qs{}{Evaluate the double integral \[
		\iint_{R} x^2 y \, dA
.\] 
where $R$ is the region bounded by  $y = x^2$ and $y = 2x$.
}

\sol{
	First, find the intersection points of $y = x^2$ and $y = 2x$ :
	 \[
	 x^2 = 2x \implies x^2 - 2x = 0 \implies x(x-2) = 0
	 .\] 
	 so $x = 0$ and $x = 2$. The region $R$ is described by  $x^2 \leq 2x$ for $0 \leq x \leq 2$.


	 set up the integral:
	 \[
	 \iint_{R} x^2 y \, dA = \int_{0}^{2} \int_{x^2}^{2x} x^2 y \, dy  \, dx 
	 .\] 

	 Evaluate the inner integral:
	 \[
		 \int_{x^2}^{2x} x^2 y \, dy = x^2 \left[ \frac{y^2}{2} \right]_{x^2}^{2x} = x^2 \left( \frac{(2x)^2}{2} - \frac{(x^2)^2}{2} \right) 
	 .\] 
	 \[
		 \boxed{= \frac{2(32)}{5} - \frac{128}{14} = \frac{64}{5} - \frac{64}{7} = \frac{448 - 320}{35} = \frac{128}{35}}
	 .\] 
}

\qs{}{Find the directional derivative of $f(x,y,z) = x^2yz^3$ at the point $P(2, 1, -1)$ in the direction of the vector $\mathbf{v} = \mathbf{i} + 2\mathbf{j} - 2\mathbf{k}$}

\sol{
	First, compute the gradient of $f$:
	 \[
		 \nabla f = \left( \frac{ \partial f }{ \partial x }, \frac{ \partial f }{ \partial y }, \frac{ \partial f }{ \partial z }    \right) = (2xyz^3, x^2z^3, 3x^2yz^2)
	.\] 

	Evaluate at $P(2, 1, -1)$:
	 \[
	\nabla f(2, 1, -1) = (2 \cdot 2 \cdot 1 \cdot (-1)^3, 2^2 \cdot (-1)^3 3 \cdot 2^2 \cdot 1 \cdot (-1)^2)
	.\] 
	\[
	 = (2 \cdot 2 \cdot 1 \cdot (-1), 4 \cdot (-1), 3 \cdot 4 \cdot 1 * 1) = (-4, -4, 12)
	.\] 

	Normalize the direction vector $\mathbf{v} = (1, 2, -2)$ :
	\[
	|\mathbf{v}| = \sqrt{1^2 + 2^2 + (-2)^2} = \sqrt{1 + 4 + 4}  = \sqrt{9}  = 3
	.\] 
	\[
		\mathbf{u} = \frac{\mathbf{v}}{|\mathbf{v}\|} = \left( \frac{1}{3}, \frac{2}{3}, - \frac{2}{3} \right)
	.\] 

	The direction derivative is:
	\[
		D_{\mathbf{u}} f = \nabla f \cdot \mathbf{u} = (-4, -4, 12) \cdot \left( \frac{1}{2}, \frac{2}{3}, - \frac{2}{3} \right) 
	.\] 
	\[
	= - \frac{4}{3} - \frac{8}{3} - \frac{24}{3} = - \frac{36}{3} = -12
	.\] 
}

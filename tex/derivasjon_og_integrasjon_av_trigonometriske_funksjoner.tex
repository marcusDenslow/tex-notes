\chapter{definisjon}
\section{reglel}
\begin{align*}
f(x) = \sin  \implies f'(x) = \cos x \\
g(x) = \cos x \implies g'(x) =  \sin x
.\end{align*}

finn $\left( \tan x \right)'$
\begin{align*}
    \left( \frac{\sin x}{\cos x}' \right)  = \left( \sin x \right)' \times \cos x - \sin x \times \left( \cos x \right)' \\
	= \frac{\cos ^2 x + \sin ^2 x}{\cos^2 x} \\
	= \frac{1}{\cos ^2 x} = 1 + \tan^2 x
.\end{align*}

regel:
\begin{align*}
    \int \sin x \, dx = -\cos x  C\\
	\int  \cos x \, dx = \sin x + C
.\end{align*}

huskeregel:
\begin{align*}
    \int tan x \, dx = \int \frac{\sin x}{\cos x} \, dx = \int - \frac{1}{u} \, du= - \int u^{-1} \, du = \ln \left| u \right|  + C = -\ln lr\left|\cos x  \right| + C
	\text{la } u = \cos x\\
	\frac{du}{dx} = -\sin x\\
	dx = - \frac{du}{\sin x}
.\end{align*}

\chapter{oppgaver}
\section{3.90}
deriverr funksjonen gitt ved\\
a) $\displaystyle f(x) = \sin 3x$
bruker kjerneregelen:
\begin{align*}
	\frac{d}{dx} \left[ f(g(x)) \right] = f'(g(x)) \times g'(x)\\
	\sin(u)' \text{ (der u = 3x) } = \cos(u)\\
	\text{derivert av u } 3x' = 3\\
	\text{ kombinerer indre og ytre } \\
	f'(x) = \cos(3x) \times 3 = 3\cos(3x)
\end{align*}

b) $\displaystyle g(x) = \frac{1}{5} \sin 5x$
\begin{align*}
	g'(x) = \frac{1}{5} \times \frac{d}{dx} \left[ \sin(5) \right] \\
	\text{Ytre funksjon: } \sin(u) \to \cos(u) \\
	\text{Indre Funksjon: } 5x \to 5\\
	\text{resultat: } \cos(5x) \times 5\cos(5x)
.\end{align*}

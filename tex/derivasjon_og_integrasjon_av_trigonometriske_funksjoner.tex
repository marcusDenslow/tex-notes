\chapter{definisjon}
\section{reglel}
\begin{align*}
f(x) = \sin  \implies f'(x) = \cos x \\
g(x) = \cos x \implies g'(x) =  \sin x
.\end{align*}

finn $\left( \tan x \right)'$
\begin{align*}
    \left( \frac{\sin x}{\cos x}' \right)  = \left( \sin x \right)' \times \cos x - \sin x \times \left( \cos x \right)' \\
	= \frac{\cos ^2 x + \sin ^2 x}{\cos^2 x} \\
	= \frac{1}{\cos ^2 x} = 1 + \tan^2 x
.\end{align*}

regel:
\begin{align*}
    \int \sin x \, dx = -\cos x  C\\
	\int  \cos x \, dx = \sin x + C
.\end{align*}

huskeregel:
\begin{align*}
    \int tan x \, dx = \int \frac{\sin x}{\cos x} \, dx = \int - \frac{1}{u} \, du= - \int u^{-1} \, du = \ln \left| u \right|  + C = -\ln lr\left|\cos x  \right| + C
	\text{la } u = \cos x\\
	\frac{du}{dx} = -\sin x\\
	dx = - \frac{du}{\sin x}
.\end{align*}

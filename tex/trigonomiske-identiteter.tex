\section{oppgave}
\qs{}{Finn $\sin (u+v)$}
\sol
\begin{align*}
    \sin(u+v) = \cos(90 - (u+v))\\
	\cos 90 \cdot \cos (u + v) \\
	= \cos ((90 - u)-v)\\
	= \cos(90 - u) \cdot \cos v + \sin(90 \degree  - u) \sin v \\
	\sin(u) \cdot \cos v + \cos u \cdot \sin v
.\end{align*}

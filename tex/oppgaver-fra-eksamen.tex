\qs {1a}{Regn ut integralet \\
	$\int x^2 \cdot \ln x \, dx$
}

\sol
\begin{align*}
	\int  u'v                  & = uv - \int uv'                                             \\
	u                          & = \frac{1}{3}x^3 \quad v = \ln x                            \\
	u'                         & = x^2 \quad v' = \frac{1}{x}                                \\
	\int x^2 \cdot \ln x \, dx & = \frac{1}{3} x^{3} \cdot \ln x - \int \frac{1}{3}x^2 \, dx \\
	                           & = \frac{1}{3}x^3 \cdot \ln x - \frac{1}{9}x^3 + C           \\
	                           & = x^3 \left( \ln x - \frac{1}{3} \right) +C
	.\end{align*}

\qs{1b}{Besten $x$ når  $\int_{0}^{x} \sin \left( \pi t + \frac{\pi}{4} \right)  \, dt = 0$ og $x \in \left<0, \pi \right>$}
\sol
\begin{equation}
	\int_{0}^{x} \sin \left( \pi t + \frac{\pi}{4} \right)  \, dt = 0  = \int_{t = 0}^{t = x} \sin \left( u \right) \frac{du}{\pi} = \frac{1}{\pi} \left[ -\cos u \right]_{t = 0}^{t = x}
	.\end{equation}
\begin{align*}
	\frac{1}{\pi} \left[ -\cos \left( \pi t + \frac{\pi}{4} \right)  \right]_{0}^{x} & = 0 \quad | \cdot \pi \\
	\frac{1}{\pi}
	.\end{align*}

	delvis integrasjon


	husk $()u \times v' = u'v + uv'$ \\
	$\int{(u\times v) dx} = \int{u'v + uv' dx}$ \\
	$uv+C = \int{u'v dx} + \int{}uv' dx$ \\
	$\int{u'v dx}= u \times v - \int{uv' dx}t$ \\
	$\int{u'v = u \times v - \int{uv'}}$ \\
	brukes når det er enklere å regne ut $\int{uv'}$ enn $\int{u'v}$


\section{eksempler}
F.eks: \\
$\int{2x \ln x dx}$
\begin{align*}
	&= la u' = 2x \text{ og } v= \ln x \\
	&= u=x^2 \quad v' = \frac{1}{x} \\
	&= x^2 \ln x - \int{x^2 \frac{1}{x}dx} = x^2 \ln x - \int{x dx} \\
	&= x^2 \ln x - \frac{1}{x}x^2 + C \\
.\end{align*}


eksempel 2

\begin{align*}
    &= \int_{0}^{1} e^{(2x)}(2x-1) \, dx  \\
	&= \text{ regner først det ubestemte itegralet } \\
	&= \int{e^{(2x)}(2x-1)}dx \\
	&= \text{la } u'=e^{(2x)} \quad v=2x-1 \\
	&= u=\frac{1}{2} e^{(2x)}, \quad v=2x-1 \\
	&= \frac{1}{2}e^{(2x)} (2x-1) - \frac{\int_1}{2}e^{(2x)}2  \, dx  \\
	&= \frac{1}{2}e^{(2x)}(2x-1) - \int e^{(2x)}  \, dx  \\
	&= \frac{1}{2}e^{(2x)}(2x-1)-\frac{1}{2}e^{(2x)}+C \\
	&= e^{(2x)}(x-1) +C \\
.\end{align*}

fullfører:
\begin{align*}
	&= \int_{0}^{1} e^{(2x)}(2x-1) \, dx = [e^{(2x)}(x-1)]_{0}^{1}=0-(-1)=1  \\
.\end{align*}

	&= x \ln x - x + C \\
